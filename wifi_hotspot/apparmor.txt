#include <tunables/global>

profile wifi_hotspot flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities needed to run an AP and manage the bridge
  capability net_admin,
  capability net_raw,
  capability net_bind_service,
  capability kill,

  # Network access: nl80211 (netlink), raw sockets, ping verification
  network netlink raw,
  network netlink dgram,
  network packet raw,
  network packet dgram,
  network inet dgram,
  network inet stream,
  network inet raw,
  network inet6 dgram,
  network inet6 stream,
  network unix dgram,
  network unix stream,

  file,
  signal (send,receive),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** ix,
  /sbin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/s6-overlay/** rix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Add-on options and bridge revert info
  /data/** rw,

  # Interface discovery
  /sys/class/net/ r,
  /sys/class/net/** r,
  /sys/devices/** r,
  @{PROC}/sys/net/** r,
  @{PROC}/*/net/** r,
  @{PROC}/*/fd/ r,
  @{PROC}/*/stat r,
  @{PROC}/*/status r,

  # Host D-Bus (NetworkManager control via nmcli)
  /run/dbus/system_bus_socket rw,

  # rfkill (best-effort radio unblock)
  /dev/rfkill rw,
}
